var namespace = function (name) {
    var names = name.split(".");
    var current = global;
    for (var i = 0; i < names.length; i++) {
        var namespace = current[names[i]];
        if (!namespace || typeof namespace !== "object") {
            current[names[i]] = namespace = {};
        }
        current = namespace;
    }
    return current;
};

define(['mediator', 'permissions'], 
function (mediator, permissions) {
    var facade = {};
    facade.subscribe = function (event, subscriber, callback) {
	if (permissions.validate(event, subscriber)) {
			    mediator.subscribe(event, subscriber, callback);
        }
    };                    
    facade.publish = function (event) {
	mediator.publish.apply(mediator, arguments);
    };
});
